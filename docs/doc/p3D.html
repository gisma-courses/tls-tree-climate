Generating ENVI-met 3DPLANT Trees from ALS Data

Overview

This tutorial demonstrates a full workflow to derive simplified ENVI-met-compatible 3DPLANT trees from airborne laser scanning (ALS) data. The process includes:

  • Voxelization of ALS point cloud data
  • Leaf Area Density (LAD) calculation per voxel column
  • Clustering of LAD profiles to define representative tree types
  • Export of tree locations as GIS vector points
  • Export of LAD profiles as ENVI-met .pld 3DPLANT definitions

Load required libraries

library(lidR)
library(terra)
library(dplyr)
library(sf)
library(here)
library(XML)
library(stats)
library(tibble)
library(rprojroot)

Set global parameters

las_file <- here("data/ALS/tiles/")
res_xy <- 1
res_z  <- 1
k      <- 0.3
scale_factor <- 1.2
crs_code <- 25832
output_gpkg <- "data/envimet/envimet_p3dtree_points.gpkg"
xml_output_file <- "data/envimet/als_envimet_trees.pld"
species_raster <- rast(sprintf("data/aerial/%s_%sm.tif", "agg_cleand", res_xy))
n_clusters <- 100

Step 1: Read and preprocess LAS data

las = merge_las_tiles(
  tile_dir = las_file,
  output_file = "data/ALS/merged_output.laz",
  chunk_size = 10000,
  workers = 6
)

crs(las) <- "EPSG:25832"
las <- normalize_height(las, knnidw(k = 6, p = 2))
las <- filter_poi(las, Z > 0)

Step 2: Voxelization

voxels <- voxel_metrics(las, ~length(Z), res = res_xy, dz = res_z)

Step 3: LAD profile calculation

lad_df <- convert_voxel_lad_long(voxels, res_z = res_z, k = k, scale_factor = scale_factor)

Step 4: Add tree species class

species_at_xy <- extract(species_raster, lad_df[, c("x", "y")])
lad_df$species_class <- species_at_xy[, 2]

Step 5: Clustering LAD profiles

lad_df$xy_key <- paste(lad_df$x, lad_df$y)
lad_matrix <- lad_df %>% 
  tidyr::pivot_wider(names_from = z, values_from = lad, values_fill = 0) %>%
  column_to_rownames("xy_key") %>%
  as.matrix()

clustering <- kmeans(lad_matrix, centers = n_clusters, nstart = 100)
lad_df$cluster <- clustering$cluster[match(lad_df$xy_key, rownames(lad_matrix))]

Step 6: Assign ENVI-met compatible IDs

cluster_ids <- unique(lad_df$cluster)
cluster_mapping <- data.frame(
  cluster = cluster_ids,
  ENVIMET_ID = sapply(cluster_ids, int_to_base36)
)
lad_df <- left_join(lad_df, cluster_mapping, by = "cluster")

Step 7: Export spatial locations

point_df <- lad_df[!duplicated(lad_df$xy_key), c("x", "y", "ENVIMET_ID")]
sf_points <- st_as_sf(point_df, coords = c("x", "y"), crs = crs_code)
st_write(sf_points, output_gpkg, delete_layer = TRUE)

Step 8: Export XML-based PLANT3D definitions

export_lad_to_envimetp3d(lad_df, file_out = xml_output_file)

Result

  • A geopackage containing tree positions and ENVIMET_IDs used for ENVI-met domain placement.
  • A .pld file with clustered LAD profiles, ready for use in ENVI-met 3DPLANT simulations.

This method allows a realistic but computationally manageable vegetation representation in ENVI-met using TLS/ALS-derived structure.
